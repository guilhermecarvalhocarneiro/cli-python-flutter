import 'dart:convert';
import 'package:flutter/foundation.dart';

class AppNameModel {
  String id;
  String nextUrl;
  String previousUrl;

  AppNameModel({this.id, this.nextUrl, this.previousUrl});

  AppNameModel.fromMap(Map<String, dynamic> json) {
    try{
      id = json['id'];
      nextUrl = json['nextUrl'];
      previousUrl = json['previousUrl'];
    } catch (error){
      debugPrint("DebugError: No método fromMap do AppNameModel  -> $error");
    }
  }

  Map<String, dynamic> toMap() {
    final Map<String, dynamic> data = new Map<String, dynamic>();
    try{
      data['id'] = this.id;
      data['nextUrl'] = this.nextUrl;
      data['previousUrl'] = this.previousUrl;
    }catch(error){
      debugPrint("DebugError: No método toMap do AppNameModel  -> $error");
    }
    return data;
  }
}
